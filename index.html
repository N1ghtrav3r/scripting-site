Bash basis

Variabelen

naam="Gunther"
echo "$naam"

getal=5
echo $((getal + 2))

Input vragen

read -p "Wat is uw naam? " naam
echo "Hallo $naam"

If / elif / else

if [ "$naam" = "Gunther" ]; then
  echo "Welkom"
elif [ "$naam" = "Admin" ]; then
  echo "Beheerder"
else
  echo "Onbekende gebruiker"
fi

Best gebruikte vergelijkingen

= gelijk aan (tekst)
!= niet gelijk aan
-eq gelijk aan (getallen)
-ne niet gelijk aan
-lt kleiner dan
-gt groter dan
-le kleiner of gelijk
-ge groter of gelijk

For loop

for i in 1 2 3 4 5; do
  echo "$i"
done

for i in {1..5}; do
  echo "$i"
done

While loop

teller=1
while [ $teller -le 5 ]; do
  echo "$teller"
  ((teller++))
done

Case

read -p "Kies start/stop: " keuze
case $keuze in
  start) echo "Starten" ;;
  stop) echo "Stoppen" ;;
  *) echo "Ongeldige keuze" ;;
esac

Exit codes

if ping -c 1 8.8.8.8 > /dev/null 2>&1; then
  echo "Netwerk OK"
  exit 0

  echo "Netwerk fout"
  exit 1
fi

[[ ]] en (( ))

if [[ "$naam" == G* ]]; then
  echo "Begint met G"
fi

x=5

  echo "Groter dan 3"
fi

IFS

read -ra lijst <<< "rood;groen;blauw"

  echo "$item"
done
IFS=$oudeIFS

Bash netwerk

Gateway ophalen

gateway=$(ip route | awk '/default/ {print $3}')
echo "$gateway"

Gateway pingen

if ping -c 1 "$gateway" > /dev/null 2>&1; then
  echo "Gateway bereikbaar"
else
  echo "Gateway niet bereikbaar"
fi

Internet check

if curl -Is http://www.google.be | head -n 1 | grep "200\|301\|302" > /dev/null; then
  echo "Internet bereikbaar"
else
  echo "Internet niet bereikbaar"
fi

Adapterstatus

ip -br link
nmcli device status

Eenvoudig netwerk-script

check_adapters() {
  ip -br link | grep '^enp0s' | grep -v 'UP'
}

check_gateway() {
  gateway=$(ip route | awk '/default/ {print $3}')
  [ -n "$gateway" ]
}

ping_gateway() {
  ping -c 1 "$gateway" > /dev/null 2>&1
}

check_internet() {
  curl -Is http://www.google.be > /dev/null 2>&1
}

check_adapters

if ! check_gateway; then
  echo "Geen gateway gevonden"
  exit 1
fi

if ! ping_gateway; then
  echo "Gateway niet bereikbaar"
  exit 1
fi

if ! check_internet; then
  echo "Internet niet bereikbaar"
  exit 1
fi

echo "Alles werkt"

Datum en exacte leeftijd

$geboortedatum = [datetime]::ParseExact("14/05/1990", "dd/MM/yyyy", $null)
$vandaag = Get-Date
$leeftijd = $vandaag.Year - $geboortedatum.Year

if ($vandaag -lt $geboortedatum.AddYears($leeftijd)) {
    $leeftijd--
}

Write-Output "Leeftijd: $leeftijd"

    Sort-Object RouteMetric |
    Select-Object -First 1).NextHop

Write-Output $gateway

Eenvoudig netwerk-script

function Test-GatewayBestaat {
    $script:gateway = (Get-NetRoute -DestinationPrefix "0.0.0.0/0" |
        Sort-Object RouteMetric |
        Select-Object -First 1).NextHop

    return -not [string]::IsNullOrWhiteSpace($script:gateway)
}

function Test-GatewayPing {
    return Test-Connection $script:gateway -Count 1 -Quiet
}

function Test-Internet {
    return (Test-NetConnection www.google.be -Port 80).TcpTestSucceeded
}

if (-not (Test-GatewayBestaat)) {
    Write-Output "Geen gateway gevonden"
    exit 1
}

if (-not (Test-GatewayPing)) {
    Write-Output "Gateway niet bereikbaar"
    exit 1
}

if (-not (Test-Internet)) {
    Write-Output "Internet niet bereikbaar"
    exit 1
}

Write-Output "Alles werkt"
