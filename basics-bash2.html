#!/bin/bash
read -p "Geef je naam: " naam

#!/bin/bash
if [ -f bestand.txt ]; then
 echo "Bestand bestaat"
else

fi

 echo "Groter dan 10"
fi

if [ "$naam" = "Gunther" ]; then
 echo "Welkom master"
fi

exit 0   # alles ok
