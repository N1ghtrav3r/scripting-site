Conditionele statements

if [ $1 -eq $2 ]; then
  echo "they are equal"
else
  echo "they are NOT equal"
fi

if [ $1 -eq $2 ]; then
  echo "they are equal"
elif [ $1 -gt $2 ]; then
  echo "first is greater than second"
elif [ $1 -lt $2 ]; then
  echo "first is smaller than second"
fi

  echo "welcome $n"
  n=$((n+1))

Loops

for i in 1 2 3 4 5
do
  echo "Welcome $i times"
done

for i in {1..5}
do
  echo "Welcome $i times"
done

for i in {0..10..2}
do
  echo "Welcome $i times"
done

for car in bmw ford toyota nissan
do
  echo "Value of car is: $car"
done

for i in $(ls /tmp/*)
do
  echo "File $i"
done
declare -i i=0
while [[ $i < 5 ]]
do
  echo $i
  ((i++))
done

declare -i i=0
until [[ $i > 5 ]]
do
  echo $i
  ((i++))
done

De les onderscheidt duidelijk for, while en until. ŅłĆfileciteŅłéturn2file7ŅłéL20-L28Ņłü
